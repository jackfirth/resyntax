test: "hash-ref with hash-set! can be simplified to hash-ref!"
------------------------------
(define h (make-hash))
(define k 'a)
(hash-ref h k (λ ()
                (define v (+ 1 2 3))
                (hash-set! h k v)
                v))
==============================
(define h (make-hash))
(define k 'a)
(or (hash-ref h k #false)
    (let ([v (+ 1 2 3)])
      (hash-set! h k v)
      v))
==============================
(define h (make-hash))
(define k 'a)
(hash-ref! h k (λ () (+ 1 2 3)))
------------------------------
